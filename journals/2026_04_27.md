## 刷新托盘图标
collapsed:: true
	- ((6909f083-1890-4e3a-916d-45be24382f31))
- 需求：Win10系统，使用节点小宝远程控制后，由于DPI差异，部分托盘图标变得模糊。重启资源管理器无效，反而使部分图标丢失。
- Stop all explorer.exe instances, delete the user's hidden IconCache.db file, and restart Explorer. 使用AI创建ahk脚本，有效 [windows - Refresh Icon Cache Without Rebooting - Super User](https://superuser.com/questions/499078/refresh-icon-cache-without-rebooting#_prevue)
  
  ``` AutoHotkey
  #Requires AutoHotkey v2.0
  #SingleInstance Force
  ;Reference: https://superuser.com/questions/499078/refresh-icon-cache-without-rebooting#_prevue
  
  ; ====== Config ======
  iconCache := EnvGet("LOCALAPPDATA") "\IconCache.db"
  killTimeout := 5000
  shellInitDelay := 1500
  
  ; ====== Main ======
  snapshots := GetExplorerWindowsSnapshot()
  
  RunWait(A_ComSpec ' /c taskkill /f /im explorer.exe',, "Hide")
  WaitProcessGone("explorer.exe", killTimeout)
  
  if FileExist(iconCache) {
      try FileDelete(iconCache)
      catch Error as e {
          MsgBox("删除失败:`n" iconCache "`n`n" e.Message, "错误", "Iconx")
          ExitApp
      }
  }
  
  Run("explorer.exe")
  Sleep(shellInitDelay)
  
  RestoreExplorerWindows(snapshots)
  
  ExitApp
  
  
  ; ====== Functions ======
  
  GetExplorerWindowsSnapshot() {
      arr := []
      shell := ComObject("Shell.Application")
  
      for w in shell.Windows {
          exe := ""
          try exe := w.FullName
          if !InStr(StrLower(exe), "\explorer.exe")
              continue
  
          path := ""
          try path := w.Document.Folder.Self.Path
          if (path = "")
              continue
  
          hwnd := 0
          try hwnd := w.HWND
          if !hwnd
              continue
  
          mm := WinGetMinMax("ahk_id " hwnd) ; -1=min, 0=normal, 1=max
          x := y := width := height := 0
  
          ; 最小化窗口取位置信息可能不稳定，只在非最小化时取
          if (mm != -1) {
              WinGetPos(&x, &y, &width, &height, "ahk_id " hwnd)
          }
  
          arr.Push(Map(
              "Path", path,
              "MinMax", mm,
              "X", x,
              "Y", y,
              "W", width,
              "H", height
          ))
      }
      return arr
  }
  
  RestoreExplorerWindows(arr) {
      for item in arr {
          path := item["Path"]
          mm   := item["MinMax"]
  
          ; 先打开对应目录
          Run('explorer.exe "' path '"')
  
          hwnd := FindExplorerWindowByPath(path, 3000)
          if !hwnd
              continue
  
          ; 先恢复到普通态再设位置大小，避免被最大化/最小化状态阻止
          try WinRestore("ahk_id " hwnd)
  
          if (item["W"] > 0 && item["H"] > 0) {
              try WinMove(item["X"], item["Y"], item["W"], item["H"], "ahk_id " hwnd)
          }
  
          ; 恢复原状态
          if (mm = 1) {
              try WinMaximize("ahk_id " hwnd)
          } else if (mm = -1) {
              try WinMinimize("ahk_id " hwnd)
          } ; 0则保持普通态
  
          Sleep(120)
      }
  }
  
  FindExplorerWindowByPath(path, timeoutMs := 2500) {
      start := A_TickCount
      normTarget := NormalizePath(path)
  
      while (A_TickCount - start < timeoutMs) {
          shell := ComObject("Shell.Application")
          for w in shell.Windows {
              exe := ""
              try exe := w.FullName
              if !InStr(StrLower(exe), "\explorer.exe")
                  continue
  
              p := ""
              try p := w.Document.Folder.Self.Path
              if (p = "")
                  continue
  
              if (NormalizePath(p) = normTarget) {
                  try return w.HWND
              }
          }
          Sleep(100)
      }
      return 0
  }
  
  NormalizePath(p) {
      p := StrLower(Trim(p))
      ; 去掉末尾反斜杠（保留盘符根如 c:\）
      if (StrLen(p) > 3 && SubStr(p, -0) = "\")
          p := SubStr(p, 1, -1)
      return p
  }
  
  WaitProcessGone(name, timeoutMs := 5000) {
      start := A_TickCount
      while ProcessExist(name) {
          if (A_TickCount - start > timeoutMs)
              break
          Sleep(100)
      }
  }
  ```
